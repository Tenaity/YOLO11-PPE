
### Windows (PowerShell)
python -m venv .venv

python -m pip install --upgrade pip

### macOS / Linux (Bash)
python3 -m venv .venv
source .venv/bin/activate
python -m pip install --upgrade pip
pip install -r requirements.txt
python main.py
